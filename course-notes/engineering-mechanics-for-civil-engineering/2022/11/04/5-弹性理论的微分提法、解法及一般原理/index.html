                          # 5.1 弹性力学问题的微分提法

                          # 平衡 (运动) 方程
                          (1)σij,j+fi=0(=ρu¨i)
                          # 几何方程
                          (2)εij=12(ui,j+uj,i)
                          # 应变协调方程
                          (3)εij,kl+εkl,ij−εik,jl−εjl,ik=0
                          # 本构方程
                          # 应变-应力公式
                          (4)εij=1+νEσij−νEσkkδij
                          # 应力-应变公式
                          (5)σij=2Gεij+λεkkδij
                          # 基本方程组
                          # 第一组
                          (6)平衡方程(1)本构方程(5)几何方程(2)
                          # 第二组
                          (7)协调方程(3)本构方程(4)平衡方程(1)
                          # 边界条件
                          # 力边界
                          σjiνj=X―i
                          # 位移边界
                          ui=u―i
