                                    # 3.1 位移和应变

                                    # 3.1.1 位移的描述

                                    # 拉格朗日描述法

                                    ui=xi(a1,a2,a3)−ai

                                    # 3.1.2 变形的描述

                                    # 格林应变张量
                                    Eij=12(∂xm∂ai∂xm∂aj−δij)=12(∂ui∂aj+∂uj∂ai+∂um∂ai∂um∂aj)u∇=∂ui∂ajeiej,∇⁡u=∂uj∂aieiejE=12(u∇+∇⁡u+∇⁡u⋅u∇)
                                    # 长度比
                                    λν=1+2ν⋅E⋅ν
                                    # 线元变形后的夹角变化
                                    cos⁡(ν′,t′)=(ν⋅t+2ν⋅E⋅t)1λνλt

                                    # 3.2 小应变张量

                                    # 3.2.1 小应变张量的定义和意义

                                    # 柯西应变张量 / 小应变张量, 应变位移公式 / 几何方程
                                    εij=12(ui,j+uj,i)ε=12(u∇+∇⁡u)
                                    # 长度比
                                    λν=1+ν⋅ε⋅ν
                                    # 工程正应变
                                    εν=ν⋅ε⋅ν
                                    # 两线元间的夹角变化
                                    cos⁡(ν′,t′)=(1−εν−εt)ν⋅t+2ν⋅ε⋅t
                                    # 工程剪应变
                                    γνt=2ενt=2ν⋅ε⋅t=2εijνitj

                                    若 ν,t 为坐标轴方向的单位矢量, 则

                                    γij=2εij(i≠j)

                                    # 3.2.2 小应变张量的性质

                                    # 转轴公式
                                    εm′n′=βm′iβn′jεij
                                    # 主应变
                                    (εij−ενδij)νj=0

                                    εν — 主值, 沿主方向 ν 的主应变

                                    # 特征方程
                                    εν3−θ1εν2+θ2εν−θ3=0
                                    # 第一、第二和第三应变不变量
                                    θ1=εii=ε1+ε2+ε3θ2=12(εiiεjj−εijεij)=ε1ε2+ε2ε3+ε3ε1θ3=eijkε1iε2jε3k=ε1ε2ε3

                                    # 3.3 刚体转动

                                    # 转动张量
                                    12(u∇−∇⁡u)=−Ω

                                    Ω — 转动张量, 反对称, 只有三个独立分量 Ω12,Ω23,Ω31. 记

                                    Ω12=ω3=12(∂u2∂x1−∂u1∂x2)Ω23=ω1=12(∂u3∂x2−∂u2∂x3)Ω31=ω2=12(∂u1∂x3−∂u3∂x1)

                                    张量 Ω 的反偶矢量

                                    ω=ω1e1+ω2e2+ω3e3−Ω⋅dx=ω×dx
