                        # 7.5 极坐标中的平面问题

                        # 平衡方程
                        ∂σr∂r+1r∂τrθ∂θ+σr−σθr+fr=0∂τrθ∂r+1r∂σθ∂θ+2τrθr+fθ=0
                        # 几何方程
                        εr=∂ur∂rεθ=1r∂vθ∂θ+urrγrθ=1r∂ur∂θ+∂vθ∂r−vθr

                        ur — 径向位移; vθ — 环向位移

                        # 本构方程
                        # 应变 - 应力
                        εr=1E(σr−νσθ)εθ=1E(σθ−νσr)γrθ=1Gτrθ
                        # 应力 - 应变
                        σr=2G1−ν(εr+νεθ)σθ=2G1−ν(εθ+νεr)τrθ=Gγrθ
                        # 协调方程
                        ∂2εθ∂r2+1r2∂2εr∂θ2−1r∂2γrθ∂r∂θ+2r∂εθ∂r−1r∂εr∂r−1r2∂γrθ∂θ=0
                        # 应力函数解法基本方程
                        (∂2∂r2+1r∂∂r+1r2∂2∂θ2)(∂2∂r2+1r∂∂r+1r2∂2∂θ2)ϕ=0
                        # 极坐标应力公式
                        σr=1r2∂2ϕ∂θ2+1r∂ϕ∂rσθ=∂2ϕ∂r2τrθ=1r2∂ϕ∂θ−1r∂2ϕ∂θ∂r=−∂∂r(1r∂ϕ∂θ)
                        # 边界条件
                        # 位移边界 Γu
                        ur=u―r,vθ=v―θ
                        # 力边界 Γσ
                        σrcos⁡(v,r)+τrθcos⁡(v,θ)=R―τrθcos⁡(v,r)+σθcos⁡(v,θ)=Θ―

                        在环向边界 (r=const) 上:

                        σr=N―r(θ),τrθ=T―r(θ)

                        在径向边界 (θ=const) 上:

                        σθ=N―θ(r),τθr=T―θ(r)

                        # 7.6 轴对称问题

                        # 应力函数解法

                        # 应力公式
                        σr=1r∂ϕ∂r,σθ=∂2ϕ∂r2,τrθ=0
                        # 通解
                        ϕ=Aln⁡r+Br2ln⁡r+Cr2+D
                        # 应力分量
                        σr=Ar2+B(1+2ln⁡r)+2Cσθ=−Ar2+B(3+2ln⁡r)+2Cτrθ=0
                        # 应变分量
                        εr=1E[(1+ν)Ar2+(1−3ν)B+2(1−ν)×Bln⁡r+2(1−ν)C]εθ=1E[−(1+ν)Ar2+(3−ν)B+2(1−ν)×Bln⁡r+2(1−ν)C]τrθ=0
                        # 位移分量
                        ur=1E[−(1+ν)Ar+2(1−ν)Brln⁡r−(1+ν)Br+2(1−ν)Cr]+Icos⁡θ+Ksin⁡θvθ=4BrθE+Hr+Kcos⁡θ−Isin⁡θ

                        限制原点的刚体位移 — I=K=0; 位移单值 — B=0; 限制刚体转动 — H=0

                        # 位移解法

                        限制原点的刚体位移和转动, 通解为

                        ur=C1r+C2r,vθ=0
